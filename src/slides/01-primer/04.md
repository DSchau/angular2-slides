## Interfaces

```typescript
export interface IDogData {
  name?: string;
  breed?: string;
  age?: number;
}

export class Dog {
  name: string;
  breed: string;
  age: number;

  constructor(data: IDogData) {
    this.name = data.name;
    this.breed = data.breed;
    this.age = data.age;
  }
}
```

```typescript
const dog = new Dog({ name: 'Bear', breed: 'Border Collie', age: 4 });
```
