Adding children
```
import { ProjectListComponent } from './project-list';

export const routes: RouterConfig = [
  {
    path: 'home',
    component: NavigationComponent,
    children: [
      {
        path: 'projects',
        component: ProjectListComponent,
        pathMatch: 'full'
      }
    ]
  },
];
```
